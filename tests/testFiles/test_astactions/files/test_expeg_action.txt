Definition >> DefinitionIdentifier {root.name = target.get_str({Spacing : ""});}
Definition >> DefinitionExpression {
    root.expression = target.get_str({
        Literal : "",
        RegularExp : "",
        Spacing : " ",
        SLASH : " ",
        AMPERSAND : " ",
        EXCLAMATION : " ",
        MUCH_GREATER_THAN : " ",
        QUESTION : " ",
        STAR : " ",
        PLUS : " ",
        OPEN : " ",
        CLOSE : " "
    });
}
MacroDefinition > MacroIdentifier {root.name = target.get_str({Spacing : ""});}
MacroDefinition > MacroExpression {root.expression = target.get_str();}

DefinitionIdentifier 
        ++ DefinitionExpression 
        >> Identifier {
    target.definition = root.get_str({Spacing : ""});
}
